  <action>
Create authMiddleware.js with verifyToken middleware function.

**verifyToken function:**
1. Get Authorization header: `const authHeader = req.headers.authorization`
2. Check header exists and starts with 'Bearer ': if not, return 401 with { success: false, error: "No token provided" }
3. Extract token: `const token = authHeader.split(' ')[1]`
4. Try to verify: `const decoded = jwt.verify(token, process.env.JWT_SECRET)`
5. Attach to request: `req.user = decoded` (contains { id, username, iat, exp })
6. Call `next()` to continue
7. Catch TokenExpiredError: return 401 with { success: false, error: "Token expired" }
8. Catch JsonWebTokenError: return 401 with { success: false, error: "Invalid token" }
9. Catch other errors: return 401 with { success: false, error: "Token verification failed" }

**Code structure:**
```javascript
const jwt = require('jsonwebtoken');

const verifyToken = (req, res, next) => {
  const authHeader = req.headers.authorization;
  
  if (!authHeader || !authHeader.startsWith('Bearer ')) {
    return res.status(401).json({ success: false, error: 'No token provided' });
  }
  
  const token = authHeader.split(' ')[1];
  
  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    req.user = decoded;
    next();
  } catch (error) {
    if (error.name === 'TokenExpiredError') {
      return res.status(401).json({ success: false, error: 'Token expired' });
    }
    if (error.name === 'JsonWebTokenError') {
      return res.status(401).json({ success: false, error: 'Invalid token' });
    }
    return res.status(401).json({ success: false, error: 'Token verification failed' });
  }
};

module.exports = { verifyToken };
```
